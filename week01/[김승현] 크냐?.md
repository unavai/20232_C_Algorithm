#include <stdio.h>

int main() {
    for(;;) {
	int x, y;
	scanf("%d %d", &x,&y);
	if (x == 0 && y ==0) return 0;
	if (x > y) printf("Yes");
	else printf("No");
    printf("\n");
        
    }
	return 0;
}
